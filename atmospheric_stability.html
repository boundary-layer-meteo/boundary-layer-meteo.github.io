import netCDF4 as nc
import pandas as pd
from bokeh.plotting import figure, gridplot, show, output_notebook
output_notebook()
Loading BokehJS ... 
nc_file = nc.Dataset('data/sups_rao_sonde_l1_any_v00_20210614164500.nc')
z = nc_file.variables['zg'][:]
T = nc_file.variables['ta'][:]
p = nc_file.variables['pa'][:]
nc_file.close()

p0 = 1e5; Rd = 287; cp = 1005
theta = T*(p0/p)**(Rd/cp)

fig1 = figure()
fig1.line(x=T, y=z, line_width=1.5)
fig1.xaxis.axis_label='T (K)'
fig1.yaxis.axis_label='z (m)'
fig2 = figure()
fig2.line(x=theta, y=z, line_width=1.5, line_color='orange')
fig2.xaxis.axis_label='θ (K)'
fig2.yaxis.axis_label='z (m)'
show(gridplot([fig1, fig2], ncols=2, width=300, height=300))
